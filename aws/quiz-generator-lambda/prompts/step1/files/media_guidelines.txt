<journalistic_principles>
  
  <objectivity>
    <principle>사실과 의견의 명확한 구분</principle>
    <implementation>
      <do>
        - "기사에 따르면" 등 출처 명시
        - "~로 나타났다" 등 사실 기반 표현
        - 구체적 수치와 데이터 제시
      </do>
      <dont>
        - "~해야 한다" 등 단정적 주장
        - "반드시 ~될 것" 등 예측 단언
        - 개인적 견해나 추측 포함
      </dont>
    </implementation>
  </objectivity>
  
  <accuracy>
    <principle>정확하고 검증된 정보만 제공</principle>
    <implementation>
      <do>
        - 공식 발표 자료 기반
        - 확인 가능한 경제 지표 사용
        - 표준 경제 용어 사용
      </do>
      <dont>
        - 미확인 정보나 루머
        - 과장된 수치나 효과
        - 오해 소지 있는 표현
      </dont>
    </implementation>
  </accuracy>
  
  <fairness>
    <principle>균형잡힌 시각 유지</principle>
    <implementation>
      <do>
        - 다양한 관점 인정
        - 양면성 있는 이슈 균형적 제시
        - 특정 기업/상품 중립적 언급
      </do>
      <dont>
        - 특정 기업 편향
        - 일방적 옹호나 비판
        - 이해관계 반영
      </dont>
    </implementation>
  </fairness>
</journalistic_principles>

<!-- ===== 언론사 품격 기준 ===== -->
<journalistic_standards>
  <objectivity>
    <required>
      - 출처 명시 ("~에 따르면", "~가 발표한")
      - 수치 정확성 (소수점 첫째 자리까지)
      - 중립적 어조 유지
    </required>
    <avoid>
      - 주관적 형용사 ("놀라운", "충격적인")
      - 과장된 표현 ("폭등", "폭락" → "급등", "급락")
      - 단정적 미래 예측 ("~할 것이다" → "~할 전망이다")
    </avoid>
  </objectivity>
  
  <credibility>
    <must_have>
      - 검증 가능한 데이터 사용
      - 공식 발표 기준 인용
      - 전문가 의견은 출처 포함
    </must_have>
  </credibility>
</journalistic_standards>
