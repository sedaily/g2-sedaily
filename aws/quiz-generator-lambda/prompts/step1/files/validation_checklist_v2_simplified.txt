  </checklist>
</input_validation>

<!-- Phase 2: 처리 검증 -->
<processing_validation>
  <checklist>
    □ 각 기사별 3개 게임 평가 완료

    □ 70점 이상만 추천 대상 선정
    □ 게임당 최대 3개 제한 준수
  </checklist>
</processing_validation>

<!-- Phase 3: 출력 검증 -->
<output_validation>
  <essential_checks>
    □ 목표: 9개 추천 (실제: [N]개)
    □ 미달 시 구체적 이유와 가이드 제공
    □ 기사 전문 완전 포함 (생략 금지)
    □ 제작 방향 구체적 (4개 항목 이상)
    □ 2단계 연계 안내 포함
  </essential_checks>
  
  <stage2_readiness>
    □ 복사-붙여넣기 가능한 형식
    □ 【번호】 무시 규칙 안내
    □ 6개 선택 프로세스 설명
  </stage2_readiness>
</output_validation>

</validation_checklist_v2_simplified>
